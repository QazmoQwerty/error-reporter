342  /* get a specific line from a text file */
343  std::string Diagnostic::getLine(std::string fileName, int line) {
344  std::fstream file(fileName);
345  file.seekg(std::ios::beg);
346  for (int i=0; i < line - 1; ++i)

348  std::string ret;
349  std::getline(file, ret);

353  /* split a string into its lines */
354  std::vector<std::string> Diagnostic::splitLines(const std::string& str) {
355  std::vector<std::string> strings;
356  std::string::size_type loc = 0;
357  std::string::size_type prev = 0;
358  while ((loc = str.find("\n", prev)) != std::string::npos) {
359  strings.push_back(str.substr(prev, loc - prev));
360  prev = loc + 1;
361  }
362  // To get the last substring (or only, if delimiter is not found)
363  strings.push_back(str.substr(prev));
364  return strings;
365  }

381  Diagnostic& Diagnostic::print(std::ostream& out) {
382  // find the maximum line (to know by how much to indent the bars)
383  auto maxLine = loc.line;
384  for (auto& secondary : secondaries)
385  if (secondary.loc.line > maxLine)
386  maxLine = secondary.loc.line;
387 
388  sortSecondaries();
389  bool showAbove = false; // if there are any messages on the line of the error, point to the error from above instead
390 
391  for (size_t i = 0; !showAbove && i < secondaries.size(); i++)
392  if (onSameLine(secondaries[i], *this))
393  showAbove = true;
394 
395  // print the main error message
396  if (msg != "")
397  out << color(tyToString() + ": ") << colors::bold(msg) << "\n";
398 
399  if (loc.file == nullptr) return *this;
400 
401  // print the file the error is in
402  printIndent(out, maxLine, false);
403  out << color("╭─ ") << loc.file->str() << color(" ─╴") << "\n";
404 
405  bool isFirst = true;
406  unsigned int lastLine = 0;
407  size_t i = 0;
408 
409  // first print all messages in the main file which come before the error
410  while (i < secondaries.size() && secondaries[i].loc.file == loc.file && secondaries[i].loc.line < loc.line) {
411  auto &secondary = secondaries[i];
412 
413  if (isFirst) {
414  printIndent(out, maxLine);
415  out << "\n";
416  isFirst = false;
417  } else if (lastLine < secondary.loc.line - 1) {
418  if (lastLine == secondary.loc.line - 2)
419  printPaddingLine(out, maxLine, secondary.loc.line - 1, secondary.loc.file);
420  else printPaddingLine(out, maxLine);
421  }
422  lastLine = secondary.loc.line;
423  std::string line = getLine(loc.file->str(), secondary.loc.line);
424  printIndentWithLineNum(out, secondary.loc.line, maxLine);
425  out << line << "\n";
426  showSecondariesOnLine(out, line, i, maxLine);
427  }
428 
429  std::string line = getLine(loc.file->str(), loc.line);
430 
431  if (isFirst && !showAbove) {
432  printIndent(out, maxLine);
433  out << "\n";
434  } else if (lastLine != 0 && lastLine < loc.line - 1) {
435  if (lastLine == loc.line - 2)
436  printPaddingLine(out, maxLine, loc.line - 1, loc.file);
437  else printPaddingLine(out, maxLine);
438  }
439  lastLine = loc.line;
440 
441  if (showAbove) {
442  if (subMsg != "") {
443  for (auto currLine : splitLines(subMsg)) {
444  printIndent(out, maxLine);
445  for (unsigned int i = 0; i < loc.start; i++)
446  out << (line[i] == '\t' ? "\t" : " ");
447  out << color(currLine);
448  out << "\n";
449  }
450  }
451 
452  printIndent(out, maxLine);
453 
454  for (unsigned int i = 0; i < loc.start; i++)
455  out << (line[i] == '\t' ? "\t" : " ");
456  for (unsigned int i = 0; i < loc.end - loc.start; i++)
457  out << color("v");
458  out << "\n";
459  }
460 
461  printIndentWithLineNum(out, maxLine);
462  out << line << "\n";
463 
464  if (!showAbove && subMsg != "") {
465  auto split = splitLines(subMsg);
466  for (size_t i = 0; i < split.size(); i++) {
467  printIndent(out, maxLine);
468  for (unsigned int i = 0; i < loc.start; i++)
469  out << (line[i] == '\t' ? "\t" : " ");
470  for (unsigned int j = 0; j < loc.end - loc.start; j++)
471  out << color(i == 0 ? "^" : " ");
472  out << " ";
473  out << color(split[i]) << "\n";
474  }
475  }
476 
477  if (i < secondaries.size() && onSameLine(secondaries[i], *this))
478  showSecondariesOnLine(out, line, i, maxLine);
479 
480  auto currFile = loc.file;
481  while (i < secondaries.size() && secondaries[i].loc.file) {
482  auto &secondary = secondaries[i];
483 
484  if (secondary.loc.file->str() != currFile->str()) {
485  currFile = secondary.loc.file;
486  for (unsigned int i = 0; i < std::to_string(maxLine).size() + 2; i++) out << color("─");
487  out << color("╯") << "\n";
488  printIndent(out, maxLine, false);
489  out << color("╭─ ") << currFile->str() << color(" ─╴") << "\n";
490  printIndent(out, maxLine);
491  out << "\n";
492  } else if (lastLine == secondary.loc.line - 2)
493  printPaddingLine(out, maxLine, secondary.loc.line - 1, secondary.loc.file);
494  else if (lastLine < secondary.loc.line - 1)
495  printPaddingLine(out, maxLine);
496 
497  lastLine = secondary.loc.line;
498  std::string line = getLine(currFile->str(), secondary.loc.line);
499  printIndentWithLineNum(out, secondary.loc.line, maxLine);
500  out << line << "\n";
501  showSecondariesOnLine(out, line, i, maxLine);
502  }
503 
504  for (unsigned int i = 0; i < std::to_string(maxLine).size() + 2; i++) out << color("─");
505  out << color("╯") << "\n";
506  for (; i < secondaries.size(); i++) {
507  auto& secondary = secondaries[i];
508  printIndent(out, maxLine, false);
509  out << secondary.color("• " + secondary.tyToString() + secondary.color(": "));
510  auto lines = splitLines(secondary.subMsg);
511  for (size_t idx = 0; idx < lines.size(); idx++) {
512  if (idx != 0) {
513  printIndent(out, maxLine, false);
514  for (size_t j = 0; j < secondary.tyToString().size() + 4; j++)
515  out << " ";
516  }
517  out << lines[idx] << "\n";
518  }
519  }
520  return *this;
521  }

525  /* show all the secondary messages on the current line */
526  void Diagnostic::showSecondariesOnLine(std::ostream& out, std::string &line, size_t &i, unsigned int maxLine) {
527  auto &first = secondaries[i];
528  printIndent(out, maxLine);
529  if (i + 1 >= secondaries.size() || !onSameLine(first, secondaries[i + 1])) {
530  // only one secondary concerning this line
531  for (size_t idx = 0; idx < first.loc.start; idx++)
532  out << (line[idx] == '\t' ? '\t' : ' ');
533  for (size_t idx = 0; idx < first.loc.end - first.loc.start; idx++)
534  out << first.color("~");
535  auto lines = splitLines(first.subMsg);
536  for (size_t idx = 0; idx < lines.size(); idx++) {
537  if (idx != 0) {
538  printIndent(out, maxLine);
539  for (size_t j = 0; j < first.loc.end; j++)
540  out << (line[j] == '\t' ? '\t' : ' ');
541  }
542  out << " " << first.color(lines[idx]) << "\n";
543  }
544  i++;
545  } else {
546  for (size_t lineIdx = 0; lineIdx < line.size(); lineIdx++) {
547  bool b = false;
548  for (auto idx = i; !b && idx < secondaries.size() && onSameLine(secondaries[idx], first); idx++)
549  if (secondaries[idx].loc.start <= lineIdx && lineIdx < secondaries[idx].loc.end) {
550  out << secondaries[idx].color("~");
551  b = true;
552  }
553  if (!b) out << " ";
554  }
555  out << "\n";
556  for (; i < secondaries.size() && onSameLine(secondaries[i], first); i++) {
557  printIndent(out, maxLine);
558  for (size_t j = 0; j < secondaries[i].loc.start; j++) {
559  bool b = false;
560  for (auto idx = i; !b && idx < secondaries.size() && onSameLine(secondaries[idx], first); idx++)
561  if (secondaries[idx].loc.start == j) {
562  out << secondaries[idx].color("│");
563  b = true;
564  }
565  if (!b) out << " ";
566  }
567  auto lines = splitLines(secondaries[i].subMsg);
568  for (size_t idx = 0; idx < lines.size(); idx++) {
569  if (idx == 0)
570  out << secondaries[i].color("╰ ") << secondaries[i].color(lines[idx]) << "\n";
571  else {
572  printIndent(out, maxLine);
573  for (size_t j = 0; j < secondaries[i].loc.start; j++) {
574  bool b = false;
575 
576  for (auto idx = i; !b && idx < secondaries.size() && onSameLine(secondaries[idx], first); idx++)
577  if (secondaries[idx].loc.start == j) {
578  out << secondaries[idx].color("│");
579  b = true;
580  }
581  if (!b) out << " ";
582  }
583  out << secondaries[i].color(" ") << secondaries[i].color(lines[idx]) << "\n";
584  }
585  }
586  }
587  }
588  }

592  /* sort the vector of secondary messages based on the order we want to be printing them */
593  void Diagnostic::sortSecondaries() {
594  auto file = loc.file;
595  std::sort(
596  std::begin(secondaries), std::end(secondaries),
597  [file](Diagnostic &a, Diagnostic &b) {
598  if (!a.loc.file)
599  return false;
600  if (!b.loc.file)
601  return true;
602  if (a.loc.file == file && b.loc.file != file)
603  return true;
604  if (a.loc.file != file && b.loc.file == file)
605  return false;
606  if (a.loc.file != b.loc.file)
607  return a.loc.file->str() < b.loc.file->str();
608  if (a.loc.line == b.loc.line)
609  return a.loc.start > b.loc.start;
610  return a.loc.line < b.loc.line;
611  }
612  );
613  }

617  /* a 'padding' line is an irrelevant line in between two other relevant lines */
618  void Diagnostic::printPaddingLine(std::ostream& out, unsigned int maxLine, unsigned int line, SourceFile *file) {
619  unsigned int targetSize = std::to_string(maxLine).size() + 2;
620  if (line == 0) {
621  switch (targetSize) {
622  case 3: out << " " << color("⋯") << "\n"; break;
623  case 4: out << " " << color("··") << "\n"; break;
624  default: out << " " << color("···") << "\n"; break;
625  }
626  } else {
627  auto str = " " + std::to_string(line) + " ";
628  while (str.size() < targetSize)
629  str += " ";
630  str += "│ ";
631  out << color(str);
632  if (file) out << getLine(file->str(), line);
633  out << "\n";
634  }
635  }
